ShowForEditAttribute Class

Examples

In the following example, the FileContents property is visible in read-only (edit) views only.

Copy
public class SampleViewModel
{
  public string ExistingFileContents { get; set; }

  [Display(Name = "UploadNewFile")]
  [DataType(DataType.Upload)]
  [ShowForEdit(false)]
  public HttpPostedFileBase FileContents { get; set; }
}
