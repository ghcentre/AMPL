DropDownListAttribute Class

Renders the model property as drop down list. The ItemContainer property of the attribute contains the name of the property or method which returns the sequence of list items.

public class SampleViewModel
{
  // 
  // This example uses method to get items for the drop down list.
  // You may use a property such as:
  //   public IEnumerable<SelectListItem> SamplePropertyItems { get { return .... } }
  // 
  [DropDownList("GetSamplePropertyItems")]
  public string SampleProperty { get; set; }

  public IEnumerable<SelectListItem> GetSamplePropertyItems()
  {
    return new[] {
      new SelectListItem() { Value = 0, Text = "Zero" },
      new SelectListItem() { Value = 1, Text = "One" }
    };
  }
}
